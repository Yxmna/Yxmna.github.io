class Coin():
    def __init__(self, x, y, super): #Création des valeurs par défault
        self.x = x #Le x donné devient le x de CE coin
        self.y = y #//
        self.get = False #Le coin pas encore attrapé
        self.super = super
    def draw(self): #On dessine le coin
        global score, u
        if self.get == False: #Si pas attrapé: on dessine
            if self.super: #si le coin est un super coin
                pygame.draw.ellipse(game,purple2,[self.x,self.y,21,23])
                pygame.draw.ellipse(game,purple1,[self.x,self.y,20,22])
                pygame.draw.ellipse(game,purple,[self.x,self.y,16,20])
            else: #si le coin est un simple coin
                pygame.draw.ellipse(game,yellow3,[self.x,self.y,21,23])
                pygame.draw.ellipse(game,yellow2,[self.x,self.y,20,22])
                pygame.draw.ellipse(game,yellow,[self.x,self.y,16,20])

            if self.y-size<y_pos<self.y+20 and self.x-size<x_pos<self.x+18:
            #si perso se trouve sur le coin:
                if self.super: #si le coin est un super coin
                    score += 10 #Il donne 10 points
                else: #si le coin est un simple coin
                    score += 1 #Il donne un point
                son.play() #Activation du son du coin
                self.get = True #Le coin est attrapé
                u = 0
            return score

def mapping():

    if y_pos>800 and act == 0:
        snake_y[0] = 0

        y_move = speed
        while w<i:
